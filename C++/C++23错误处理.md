```cpp
// expected.cpp

#include <expected>
#include <variant>
#include <string>

// Error cases
struct UserNotFound {};
struct DatabaseError {};
struct PermissionDenied {};
struct InvalidSettings {};

// Define the Error variant
using Error = std::variant<
    UserNotFound,
    DatabaseError,
    PermissionDenied,
    InvalidSettings
>;

// Forward declarations for types used
struct User {};
struct Profile {};
struct Settings {};
struct Config {};

// Helper functions declarations
std::expected<User, Error> loadUser(int userId);
std::expected<Profile, Error> loadProfile(const User& user);
std::expected<Settings, Error> loadSettings(const Profile& profile);
Config makeConfig(const Settings& settings);

// Composition
std::expected<Config, Error> buildConfig(int userId) {
    return loadUser(userId)
        .and_then(loadProfile)
        .and_then(loadSettings)
        .transform(makeConfig);
}
```
